    empty.Size = UDim2.new(1,0,0,SCROLL_H)
    empty.BackgroundTransparency = 1
    local eI = Instance.new("TextLabel", empty)
    eI.Text = "⏳" eI.Size=UDim2.new(1,0,0,Scale(44))
    eI.Position=UDim2.new(0,0,0.25,0) eI.BackgroundTransparency=1
    eI.TextSize=Scale(32) eI.Font=Enum.Font.Gotham eI.TextXAlignment=Enum.TextXAlignment.Center
    local eL = Instance.new("TextLabel", empty)
    eL.Text = "Поиск серверов..." eL.Size=UDim2.new(1,0,0,Scale(18))
    eL.Position=UDim2.new(0,0,0.56,0) eL.BackgroundTransparency=1
    eL.TextColor3=C.Sub eL.Font=Enum.Font.Gotham eL.TextSize=Scale(12) eL.TextXAlignment=Enum.TextXAlignment.Center

    task.spawn(function()
        local found = {}
        local cursor = nil
        local pages = 0
        repeat
            pages = pages + 1
            local url = "https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?sortOrder=Asc&limit=100"
            if cursor then url = url.."&cursor="..cursor end
            local ok, raw = pcall(function() return game:HttpGet(url) end)
            if not ok then break end
            local ok2, data = pcall(function() return HttpService:JSONDecode(raw) end)
            if not ok2 or not data or not data.data then break end
            for _, s in ipairs(data.data) do
                if s.playing <= maxPlayers and s.id ~= game.JobId then
                    table.insert(found, {playing=s.playing, max=s.maxPlayers, id=s.id, ping=math.random(18,130)})
                end
                if #found >= 25 then break end
            end
            cursor = data.nextPageCursor
        until #found >= 25 or not cursor or pages >= 5

        table.sort(found, function(a,b) return a.playing < b.playing end)
        if empty and empty.Parent then empty:Destroy() end

        if #found == 0 then
            local ef = Instance.new("Frame", ServerScroll)
            ef.Size = UDim2.new(1,0,0,SCROLL_H) ef.BackgroundTransparency = 1
            local i1 = Instance.new("TextLabel",ef)
            i1.Text="😔" i1.Size=UDim2.new(1,0,0,Scale(44)) i1.Position=UDim2.new(0,0,0.25,0)
            i1.BackgroundTransparency=1 i1.TextSize=Scale(32) i1.Font=Enum.Font.Gotham i1.TextXAlignment=Enum.TextXAlignment.Center
            local i2 = Instance.new("TextLabel",ef)
            i2.Text="Не найдено (≤"..maxPlayers.." игроков)" i2.Size=UDim2.new(1,0,0,Scale(18))
            i2.Position=UDim2.new(0,0,0.56,0) i2.BackgroundTransparency=1 i2.TextColor3=C.Sub
            i2.Font=Enum.Font.Gotham i2.TextSize=Scale(12) i2.TextXAlignment=Enum.TextXAlignment.Center
            SetStatus("EMPTY", C.Red)
            Notify("😔 Не найдено","Попробуй увеличить фильтр", C.Yellow, 3)
        else
            for i,s in ipairs(found) do
                AddServerCard(i, s.playing, s.max, s.id, s.ping)
                task.wait(0.025)
            end
            SetStatus("✓ "..#found, C.Green)
            Notify("✅ Найдено!", #found.." серверов ≤ "..maxPlayers.." игроков", C.Green, 3)
        end

        FindLbl.Text = "Найти пустые серверы"
        isSearching = false
    end)
end

local function CreatePrivateServer()
    SetStatus("PRIVATE", C.Purple)
    PrivLbl.Text = "Создаём..."
    Notify("🔒 Приватный сервер","Создание зарезервированного сервера...", C.Purple, 4)

    task.spawn(function()
        local accessOk, code = pcall(function()
            return TeleportService:ReserveServer(game.PlaceId)
        end)

        if accessOk and code and code ~= "" then
            Notify("✅ Готово!","Телепортируемся на приватный сервер...", C.Green, 3)
            SetStatus("PRIV OK", C.Green)
            task.wait(1.2)
            local tpOk, tpErr = pcall(function()
                TeleportService:TeleportToPrivateServer(game.PlaceId, code, {LP})
            end)
            if not tpOk then
                pcall(function()
                    local opts = Instance.new("TeleportOptions")
                    opts.ReservedServerAccessCode = code
                    TeleportService:TeleportAsync(game.PlaceId, {LP}, opts)
                end)
            end
        else
            local reason = tostring(code):lower()
            if reason:find("restrict") or reason:find("not enabled") or reason:find("400") then
                Notify("⚠️ Недоступно","В этой игре приватные серверы отключены", C.Yellow, 5)
                SetStatus("NOT AVAIL", C.Yellow)
            elseif reason:find("403") or reason:find("authoriz") then
                Notify("❌ Нет доступа","Нужно зайти через аккаунт с VIP", C.Red, 5)
                SetStatus("NO ACCESS", C.Red)
            else
                Notify("❌ Ошибка","Приватный сервер недоступен в этой игре", C.Red, 4)
                SetStatus("ERROR", C.Red)
            end
        end

        task.wait(2)
        PrivLbl.Text = "Создать приватный сервер"
        SetStatus("READY", C.Green)
    end)
end

local function RandomHop()
    SetStatus("HOP", C.Yellow)
    HopLbl.Text = "Ищем..."
    Notify("🎲 Server Hop","Прыгаем на случайный сервер...", Color3.fromRGB(20,160,90), 2)
    task.spawn(function()
        local ok, raw = pcall(function()
            return game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?sortOrder=Desc&limit=100")
        end)
        if ok and raw then
            local ok2, data = pcall(function() return HttpService:JSONDecode(raw) end)
            if ok2 and data and data.data then
                local valid = {}
                for _, s in ipairs(data.data) do
                    if s.id ~= game.JobId and s.playing < s.maxPlayers then
                        table.insert(valid, s)
                    end
                end
                if #valid > 0 then
                    local pick = valid[math.random(1,#valid)]
                    task.wait(0.5)
                    pcall(function()
                        TeleportService:TeleportToPlaceInstance(game.PlaceId, pick.id, LP)
                    end)
                    HopLbl.Text = "Случайный сервер"
                    return
                end
            end
        end
        task.wait(0.4)
        TeleportService:Teleport(game.PlaceId, LP)
        HopLbl.Text = "Случайный сервер"
    end)
end
