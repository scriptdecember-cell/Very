local function Tw(o,p,t,s,d)
    TweenService:Create(o, TweenInfo.new(t or 0.2,
        s or Enum.EasingStyle.Quart,
        d or Enum.EasingDirection.Out), p):Play()
end
local function Rnd(p,r) local c=Instance.new("UICorner") c.CornerRadius=UDim.new(0,r or 8) c.Parent=p return c end
local function Stk(p,col,th) local s=Instance.new("UIStroke") s.Color=col s.Thickness=th or 1 s.Parent=p return s end
local function Pad(p,v) local u=Instance.new("UIPadding") u.PaddingLeft=UDim.new(0,v) u.PaddingRight=UDim.new(0,v) u.PaddingTop=UDim.new(0,v) u.PaddingBottom=UDim.new(0,v) u.Parent=p end
local function Lbl(parent,text,size,color,font,xa,ya)
    local l=Instance.new("TextLabel",parent)
    l.Text=text l.TextSize=size or 13
    l.TextColor3=color or C.Text
    l.Font=font or Enum.Font.GothamBold
    l.BackgroundTransparency=1
    l.Size=UDim2.new(1,0,1,0)
    l.TextXAlignment=xa or Enum.TextXAlignment.Left
    l.TextYAlignment=ya or Enum.TextYAlignment.Center
    return l
end

local isSearching = false

local function SearchServers()
    if isSearching then return end
    isSearching = true
    ClearServers()
    EmptyFrame.Visible = true
    EmptyLbl.Text = "Поиск серверов..."
    EmptyIcon.Text = "⏳"
    SetStatus("SEARCH", C.Yellow)
    FindBtnLbl.Text = "Ищем..."

    local found = {}
    local cursor = nil
    local pages = 0

    task.spawn(function()
        repeat
            pages = pages + 1
            local url = "https://games.roblox.com/v1/games/"..game.PlaceId..
                "/servers/Public?sortOrder=Asc&limit=100"
            if cursor then url = url .. "&cursor=" .. cursor end

            local ok, raw = pcall(function()
                return game:HttpGet(url)
            end)
            if not ok then break end

            local ok2, data = pcall(function()
                return HttpService:JSONDecode(raw)
            end)
            if not ok2 or not data or not data.data then break end

            for _, s in ipairs(data.data) do
                if s.playing <= maxPlayers and s.id ~= game.JobId then
                    local ping = math.random(20, 120)
                    table.insert(found, {
                        playing = s.playing,
                        max     = s.maxPlayers,
                        id      = s.id,
                        ping    = ping,
                    })
                end
                if #found >= 20 then break end
            end

            cursor = data.nextPageCursor
        until #found >= 20 or not cursor or pages >= 5

        table.sort(found, function(a,b) return a.playing < b.playing end)

        EmptyFrame.Visible = #found == 0

        if #found == 0 then
            EmptyLbl.Text = "Серверы не найдены (≤"..maxPlayers.." игроков)"
            EmptyIcon.Text = "😔"
            SetStatus("EMPTY", C.Red)
            Notify("😔 Не найдено","Попробуй увеличить фильтр", C.Yellow, 3)
        else
            EmptyFrame.Visible = false
            for i, s in ipairs(found) do
                AddServerCard(i, s.playing, s.max, s.id, s.ping)
                task.wait(0.03)
            end
            SetStatus("FOUND "..#found, C.Green)
            Notify("✅ Найдено!", #found.." серверов с ≤"..maxPlayers.." игроков", C.Green, 3)
        end

        FindBtnLbl.Text = "Найти пустые серверы"
        isSearching = false
    end)
end

local function CreatePrivateServer()
    SetStatus("PRIVATE", C.Purple)
    Notify("🔒 Приватный сервер","Создаём зарезервированный сервер...", C.Purple, 3)
    task.spawn(function()
        local ok, code = pcall(function()
            return TeleportService:ReserveServer(game.PlaceId)
        end)
        if ok and code then
            Notify("✅ Сервер создан","Телепортируемся...", C.Green, 3)
            task.wait(1)
            pcall(function()
                TeleportService:TeleportToPrivateServer(game.PlaceId, code, {LP})
            end)
        else
            Notify("❌ Ошибка","Не удалось создать сервер", C.Red, 3)
            SetStatus("ERROR", C.Red)
        end
    end)
end

local function RandomHop()
    SetStatus("HOPPING", C.Yellow)
    Notify("🎲 Server Hop","Прыгаем на случайный сервер...", Color3.fromRGB(30,150,80), 2)
    task.spawn(function()
        local ok, raw = pcall(function()
            return game:HttpGet("https://games.roblox.com/v1/games/"..game.PlaceId..
                "/servers/Public?sortOrder=Desc&limit=100")
        end)
        if ok and raw then
            local ok2, data = pcall(function() return HttpService:JSONDecode(raw) end)
            if ok2 and data and data.data then
                local valid = {}
                for _, s in ipairs(data.data) do
                    if s.id ~= game.JobId and s.playing < s.maxPlayers then
                        table.insert(valid, s)
                    end
                end
                if #valid > 0 then
                    local pick = valid[math.random(1,#valid)]
                    task.wait(0.5)
                    pcall(function()
                        TeleportService:TeleportToPlaceInstance(game.PlaceId, pick.id, LP)
                    end)
                    return
                end
            end
        end
        task.wait(0.5)
        TeleportService:Teleport(game.PlaceId, LP)
    end)
end
